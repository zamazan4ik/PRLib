32  cv::Mat gray;
33  if (inputImage.channels() == 3)

36  }
37  else

39  gray = inputImage;

42  cv::Mat bw;
43  // TODO: Try to use another binarization here
44  cv::adaptiveThreshold(~gray, bw, 255, CV_ADAPTIVE_THRESH_MEAN_C, CV_THRESH_BINARY, 15, -2);
45 
46  // Create the images that will use to extract the horizonta and vertical lines
47  cv::Mat horizontal = bw.clone();
48  cv::Mat vertical = bw.clone();
49 
50  // Specify size on horizontal axis
51  int horizontalsize = horizontal.cols / 30;
52 
53  // Create structure element for extracting horizontal lines through morphology operations
54  cv::Mat horizontalStructure = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(horizontalsize,1));
55 
56  // Apply morphology operations
57  cv::erode(horizontal, horizontal, horizontalStructure, cv::Point(-1, -1));
58  cv::dilate(horizontal, horizontal, horizontalStructure, cv::Point(-1, -1));
59 
60  // Specify size on vertical axis
61  int verticalsize = vertical.rows / 30;
62 
63  // Create structure element for extracting vertical lines through morphology operations
64  cv::Mat verticalStructure = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(1, verticalsize));
65 
66  // Apply morphology operations
67  cv::erode(vertical, vertical, verticalStructure, cv::Point(-1, -1));
68  cv::dilate(vertical, vertical, verticalStructure, cv::Point(-1, -1));
69 
70  outputImage = bw.clone();
71  outputImage = outputImage - horizontal;
72  outputImage = outputImage - vertical;
73 
74  cv::bitwise_not(outputImage, outputImage);
75 }
